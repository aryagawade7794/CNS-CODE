#----------------app.py--------

from flask import Flask, render_template, request, redirect, url_for, flash, session
from flask_bcrypt import Bcrypt
import sqlite3, os

app = Flask(__name__)
app.secret_key = "secret123"
bcrypt = Bcrypt(app)

DB_PATH = "users.db"

# --- Create table if not exists ---
def init_db():
    conn = sqlite3.connect(DB_PATH)
    conn.execute("""CREATE TABLE IF NOT EXISTS users(
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        username TEXT UNIQUE,
        password TEXT
    )""")
    conn.close()

init_db()

# --- Dictionary to track login attempts ---
login_attempts = {}

@app.route('/')
def home():
    return render_template('index.html')

@app.route('/register', methods=['GET','POST'])
def register():
    if request.method == 'POST':
        user = request.form['username']
        pw = request.form['password']
        hashed = bcrypt.generate_password_hash(pw).decode('utf-8')
        try:
            conn = sqlite3.connect(DB_PATH)
            conn.execute("INSERT INTO users(username,password) VALUES(?,?)", (user, hashed))
            conn.commit()
            conn.close()
            flash("Registered successfully! Please log in.", "success")
            return redirect(url_for('login'))
        except:
            flash("Username already exists!", "danger")
    return render_template('register.html')

@app.route('/login', methods=['GET','POST'])
def login():
    if request.method == 'POST':
        user = request.form['username']
        pw = request.form['password']

        # Initialize user attempt count if not exists
        if user not in login_attempts:
            login_attempts[user] = 0

        # Check if user exceeded attempts
        if login_attempts[user] >= 3:
            flash("Too many failed attempts! Please try again later.", "danger")
            return render_template('login.html')

        conn = sqlite3.connect(DB_PATH)
        cur = conn.cursor()
        cur.execute("SELECT password FROM users WHERE username=?", (user,))
        row = cur.fetchone()
        conn.close()

        if row and bcrypt.check_password_hash(row[0], pw):
            session['username'] = user
            login_attempts[user] = 0  # reset on successful login
            flash("Login successful!", "success")
            return redirect(url_for('dashboard'))
        else:
            login_attempts[user] += 1
            remaining = 3 - login_attempts[user]
            if remaining > 0:
                flash(f"Invalid username or password! {remaining} attempt(s) left.", "danger")
            else:
                flash("Account locked! Too many failed attempts.", "danger")

    return render_template('login.html')

@app.route('/dashboard')
def dashboard():
    if 'username' in session:
        return render_template('dashboard.html', user=session['username'])
    return redirect(url_for('login'))

@app.route('/logout')
def logout():
    session.pop('username', None)
    flash("Logged out!", "info")
    return redirect(url_for('home'))

if __name__ == '__main__':
    app.run(debug=True)
